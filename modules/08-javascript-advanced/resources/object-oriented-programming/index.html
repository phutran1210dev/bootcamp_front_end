Giới thiệu OOP

Tìm hiểu về các khái niệm cơ bản của lập trình hướng đối tượng

Class (Lớp)

Template để tạo ra các object với cùng cấu trúc và hành vi

Object (Đối tượng)

Instance của class, chứa dữ liệu (properties) và hành vi (methods)

Inheritance (Kế thừa)

Class con có thể kế thừa properties và methods từ class cha

Encapsulation (Đóng gói)

Che giấu chi tiết triển khai, chỉ expose interface cần thiết

Polymorphism (Đa hình)

Cùng một method có thể hoạt động khác nhau ở các class khác nhau

Abstraction (Trừu tượng)

Ẩn đi những chi tiết phức tạp, chỉ hiển thị những gì cần thiết

Học cách tạo và sử dụng classes trong ES6+

1. Cú pháp cơ bản

JavaScript 
// Định nghĩa class
class Person {
    // Constructor - phương thức khởi tạo
    constructor(name, age, email) {
        this.name = name;      // Property (thuộc tính)
        this.age = age;
        this.email = email;
    }

    // Method (phương thức)
    introduce() {
        return `Xin chào, tôi là ${this.name}, ${this.age} tuổi`;
    }

    // Getter
    get info() {
        return `${this.name} - ${this.email}`;
    }

    // Setter
    set updateAge(newAge) {
        if (newAge > 0 && newAge < 150) {
            this.age = newAge;
        }
    }

    // Static method
    static createAdult(name, email) {
        return new Person(name, 18, email);
    }
}

// Sử dụng class
const person1 = new Person("Nguyễn Văn A", 25, "vana@email.com");
console.log(person1.introduce()); // "Xin chào, tôi là Nguyễn Văn A, 25 tuổi"
console.log(person1.info); // "Nguyễn Văn A - vana@email.com"

person1.updateAge = 26;
console.log(person1.age); // 26

const adult = Person.createAdult("Trần Thị B", "thib@email.com");
console.log(adult.age); // 18

2. Private Fields và Methods (ES2022)

JavaScript 
class BankAccount {
    // Private fields (bắt đầu với #)
    #balance = 0;
    #accountNumber;
    #pin;

    constructor(accountNumber, initialBalance = 0, pin) {
        this.#accountNumber = accountNumber;
        this.#balance = initialBalance;
        this.#pin = pin;
    }

    // Public method
    deposit(amount) {
        if (amount > 0) {
            this.#balance += amount;
            return `Đã nạp ${amount}. Số dư: ${this.#balance}`;
        }
        return "Số tiền không hợp lệ";
    }

    withdraw(amount, pin) {
        if (!this.#validatePin(pin)) {
            return "Mã PIN không đúng";
        }
        
        if (amount > 0 && amount <= this.#balance) {
            this.#balance -= amount;
            return `Đã rút ${amount}. Số dư: ${this.#balance}`;
        }
        return "Số tiền không hợp lệ hoặc không đủ số dư";
    }

    // Private method
    #validatePin(pin) {
        return this.#pin === pin;
    }

    // Getter cho balance (chỉ đọc)
    get balance() {
        return this.#balance;
    }

    get accountInfo() {
        return `Tài khoản: ${this.#accountNumber}`;
    }
}

// Sử dụng
const account = new BankAccount("123456789", 1000, "1234");
console.log(account.deposit(500)); // "Đã nạp 500. Số dư: 1500"
console.log(account.withdraw(200, "1234")); // "Đã rút 200. Số dư: 1300"
console.log(account.balance); // 1300

// Không thể truy cập private fields
// console.log(account.#balance); // SyntaxError!

Inheritance (Kế thừa)

Tạo class con kế thừa từ class cha

1. Kế thừa cơ bản với extends

JavaScript 
// Class cha (Parent/Base class)
class Animal {
    constructor(name, species) {
        this.name = name;
        this.species = species;
        this.energy = 100;
    }

    eat(food) {
        this.energy += 10;
        return `${this.name} đã ăn ${food}. Năng lượng: ${this.energy}`;
    }

    sleep() {
        this.energy += 20;
        return `${this.name} đã ngủ. Năng lượng: ${this.energy}`;
    }

    makeSound() {
        return `${this.name} phát ra âm thanh`;
    }

    getInfo() {
        return `Tên: ${this.name}, Loài: ${this.species}, Năng lượng: ${this.energy}`;
    }
}

// Class con (Child/Derived class)
class Dog extends Animal {
    constructor(name, breed) {
        super(name, "Chó"); // Gọi constructor của class cha
        this.breed = breed;
        this.loyalty = 100;
    }

    // Override method của class cha
    makeSound() {
        return `${this.name} sủa: Gâu gâu!`;
    }

    // Method riêng của class con
    wagTail() {
        this.loyalty += 5;
        return `${this.name} vẫy đuôi vui vẻ! Lòng trung thành: ${this.loyalty}`;
    }

    fetch(item) {
        this.energy -= 10;
        this.loyalty += 10;
        return `${this.name} đã đi lấy ${item}! Năng lượng: ${this.energy}, Lòng trung thành: ${this.loyalty}`;
    }

    // Override getInfo để thêm thông tin breed
    getInfo() {
        return super.getInfo() + `, Giống: ${this.breed}, Lòng trung thành: ${this.loyalty}`;
    }
}

class Cat extends Animal {
    constructor(name, color) {
        super(name, "Mèo");
        this.color = color;
        this.independence = 80;
    }

    makeSound() {
        return `${this.name} kêu: Meo meo!`;
    }

    climb() {
        this.energy -= 5;
        return `${this.name} đã trèo lên cao! Năng lượng: ${this.energy}`;
    }

    hunt() {
        this.energy -= 15;
        this.independence += 5;
        return `${this.name} đã đi săn! Năng lượng: ${this.energy}, Độc lập: ${this.independence}`;
    }

    getInfo() {
        return super.getInfo() + `, Màu: ${this.color}, Độc lập: ${this.independence}`;
    }
}

// Sử dụng
const myDog = new Dog("Buddy", "Golden Retriever");
const myCat = new Cat("Whiskers", "Xám");

console.log(myDog.makeSound()); // "Buddy sủa: Gâu gâu!"
console.log(myCat.makeSound()); // "Whiskers kêu: Meo meo!"

console.log(myDog.fetch("bóng")); // "Buddy đã đi lấy bóng! Năng lượng: 90, Lòng trung thành: 110"
console.log(myCat.hunt()); // "Whiskers đã đi săn! Năng lượng: 85, Độc lập: 85"

Encapsulation (Đóng gói)

Bảo vệ dữ liệu và kiểm soát truy cập

1. Private Properties và Methods

JavaScript 
class Car {
    // Private fields
    #engine;
    #fuel;
    #maxFuel;
    #isRunning = false;

    constructor(brand, model, maxFuel = 50) {
        this.brand = brand;
        this.model = model;
        this.#maxFuel = maxFuel;
        this.#fuel = 0;
        this.#engine = {
            temperature: 20,
            rpm: 0
        };
    }

    // Public methods
    start() {
        if (this.#isRunning) {
            return "Xe đã được khởi động rồi!";
        }
        
        if (this.#fuel <= 0) {
            return "Không thể khởi động - hết xăng!";
        }

        this.#isRunning = true;
        this.#engine.rpm = 800;
        this.#warmUpEngine();
        return `${this.brand} ${this.model} đã khởi động thành công!`;
    }

    stop() {
        if (!this.#isRunning) {
            return "Xe đã tắt rồi!";
        }

        this.#isRunning = false;
        this.#engine.rpm = 0;
        this.#coolDownEngine();
        return `${this.brand} ${this.model} đã tắt máy.`;
    }

    addFuel(amount) {
        if (amount <= 0) {
            return "Số lượng xăng phải lớn hơn 0!";
        }

        const canAdd = Math.min(amount, this.#maxFuel - this.#fuel);
        this.#fuel += canAdd;
        
        if (canAdd < amount) {
            return `Đã thêm ${canAdd}L xăng. Bình xăng đã đầy! (${this.#fuel}/${this.#maxFuel}L)`;
        }
        
        return `Đã thêm ${canAdd}L xăng. Tổng: ${this.#fuel}L`;
    }

    drive(distance) {
        if (!this.#isRunning) {
            return "Hãy khởi động xe trước!";
        }

        const fuelNeeded = distance * 0.1; // 0.1L/km
        
        if (fuelNeeded > this.#fuel) {
            return `Không đủ xăng! Cần ${fuelNeeded}L nhưng chỉ có ${this.#fuel}L`;
        }

        this.#fuel -= fuelNeeded;
        this.#updateEngineLoad(distance);
        
        return `Đã lái ${distance}km. Xăng còn lại: ${this.#fuel.toFixed(1)}L`;
    }

    // Private methods
    #warmUpEngine() {
        this.#engine.temperature = 90;
    }

    #coolDownEngine() {
        this.#engine.temperature = 20;
    }

    #updateEngineLoad(distance) {
        const load = Math.min(3000, 800 + (distance * 10));
        this.#engine.rpm = load;
        this.#engine.temperature = Math.min(110, 90 + (distance * 0.5));
    }

    // Getters for controlled access
    get fuelLevel() {
        return `${this.#fuel.toFixed(1)}/${this.#maxFuel}L`;
    }

    get engineStatus() {
        return {
            running: this.#isRunning,
            temperature: this.#engine.temperature,
            rpm: this.#engine.rpm
        };
    }

    get carInfo() {
        return `${this.brand} ${this.model} - ${this.#isRunning ? 'Đang chạy' : 'Đã tắt máy'}`;
    }
}

// Sử dụng
const myCar = new Car("Toyota", "Camry", 60);
console.log(myCar.addFuel(40)); // "Đã thêm 40L xăng. Tổng: 40L"
console.log(myCar.start()); // "Toyota Camry đã khởi động thành công!"
console.log(myCar.drive(100)); // "Đã lái 100km. Xăng còn lại: 30.0L"
console.log(myCar.fuelLevel); // "30.0/60L"

// Không thể truy cập private fields
// console.log(myCar.#fuel); // SyntaxError!

Polymorphism (Đa hình)

Cùng một interface, nhiều cách triển khai khác nhau

1. Method Overriding và Polymorphism

JavaScript 
// Base class
class Shape {
    constructor(name) {
        this.name = name;
    }

    // Method sẽ được override bởi các class con
    calculateArea() {
        throw new Error("calculateArea() phải được implement bởi class con");
    }

    calculatePerimeter() {
        throw new Error("calculatePerimeter() phải được implement bởi class con");
    }

    // Common method cho tất cả shapes
    getInfo() {
        return `${this.name} - Diện tích: ${this.calculateArea()}, Chu vi: ${this.calculatePerimeter()}`;
    }

    // Static method để tạo shape từ type
    static createShape(type, ...params) {
        switch(type.toLowerCase()) {
            case 'rectangle':
                return new Rectangle(params[0], params[1]);
            case 'circle':
                return new Circle(params[0]);
            case 'triangle':
                return new Triangle(params[0], params[1], params[2]);
            default:
                throw new Error(`Unknown shape type: ${type}`);
        }
    }
}

class Rectangle extends Shape {
    constructor(width, height) {
        super("Hình chữ nhật");
        this.width = width;
        this.height = height;
    }

    calculateArea() {
        return this.width * this.height;
    }

    calculatePerimeter() {
        return 2 * (this.width + this.height);
    }

    // Method riêng của Rectangle
    isSquare() {
        return this.width === this.height;
    }
}

class Circle extends Shape {
    constructor(radius) {
        super("Hình tròn");
        this.radius = radius;
    }

    calculateArea() {
        return Math.PI * this.radius * this.radius;
    }

    calculatePerimeter() {
        return 2 * Math.PI * this.radius;
    }

    // Method riêng của Circle
    getDiameter() {
        return 2 * this.radius;
    }
}

class Triangle extends Shape {
    constructor(side1, side2, side3) {
        super("Tam giác");
        this.side1 = side1;
        this.side2 = side2;
        this.side3 = side3;
    }

    calculateArea() {
        // Sử dụng công thức Heron
        const s = this.calculatePerimeter() / 2;
        return Math.sqrt(s * (s - this.side1) * (s - this.side2) * (s - this.side3));
    }

    calculatePerimeter() {
        return this.side1 + this.side2 + this.side3;
    }

    // Method riêng của Triangle
    isValidTriangle() {
        return (this.side1 + this.side2 > this.side3) &&
               (this.side1 + this.side3 > this.side2) &&
               (this.side2 + this.side3 > this.side1);
    }
}

// Demonstration of Polymorphism
class ShapeCalculator {
    constructor() {
        this.shapes = [];
    }

    addShape(shape) {
        if (shape instanceof Shape) {
            this.shapes.push(shape);
            return `Đã thêm ${shape.name}`;
        }
        throw new Error("Object phải là instance của Shape");
    }

    calculateTotalArea() {
        // Polymorphism: gọi calculateArea() trên các object khác nhau
        // nhưng mỗi object sẽ thực hiện theo cách riêng của nó
        return this.shapes.reduce((total, shape) => total + shape.calculateArea(), 0);
    }

    calculateTotalPerimeter() {
        return this.shapes.reduce((total, shape) => total + shape.calculatePerimeter(), 0);
    }

    getShapesSummary() {
        return this.shapes.map(shape => shape.getInfo()).join('\n');
    }

    // Method sử dụng polymorphism
    processShapes() {
        const results = [];
        
        this.shapes.forEach(shape => {
            // Tất cả shapes đều có calculateArea() và calculatePerimeter()
            // nhưng implementation khác nhau
            const area = shape.calculateArea();
            const perimeter = shape.calculatePerimeter();
            
            results.push({
                name: shape.name,
                area: area.toFixed(2),
                perimeter: perimeter.toFixed(2)
            });
        });
        
        return results;
    }
}

// Sử dụng Polymorphism
const calculator = new ShapeCalculator();

// Tạo các shapes khác nhau
const rectangle = new Rectangle(5, 3);
const circle = new Circle(4);
const triangle = new Triangle(3, 4, 5);

// Thêm vào calculator
calculator.addShape(rectangle);
calculator.addShape(circle);
calculator.addShape(triangle);

// Polymorphism in action
console.log("Tổng diện tích:", calculator.calculateTotalArea().toFixed(2));
console.log("Tổng chu vi:", calculator.calculateTotalPerimeter().toFixed(2));
console.log("\nThông tin các hình:");
console.log(calculator.getShapesSummary());

Khó
